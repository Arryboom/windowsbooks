[Wdm1.Files.Driver]
Wdm1.sys
;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Windows 2000

[Wdm1.Install.NT×86]
CopyFiles=Wdm1.Files.Driver.NT×86

[Wdm1.Files.Driver.NT×86]
Wdm1.sys,,,%COPYFLG_NOSKIP%

[Wdm1.Install.NT×86.Services]
AddService=Wdm1, %SPSVCINST_ASSOCSERVICE%, Wdm1.Service

[Wdm1.Service]
DisplayName    = %Wdm1.ServiceName%
ServiceType    = %SERVICE_KERNEL_DRIVER%
StartType      = %SERVICE_DEMAND_START%
ErrorControl   = %SERVICE_ERROR_NORMAL%
ServiceBinary  = %10%\System32\Drivers\Wdm1.sys
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;Strings

[Strings]
WDMBook="WDM Book"
Wdm1="WDM Book: Wdm1 Example, free build"
Wdm1.ServiceName="WDM Book Wdm1 Driver''

SPSVCINST_ASSOCSERVICE=0×00000002 ; Driver service is associated with device being
installed
COPYFLG_NOSKIP=2 ; Do not allow user to skip file
SERVICE_KERNEL_DRIVER=1
SERVICE AUTO_START=2
SERVICE_DEMAND_START=3
SERVICE_ERROR_NORMAL=1

DestinationDirs Section
Each line in the DestinationDirs section specifies the base directory for files listed in a CopyFi1es section.
